<script>
/* ---------- Universal editor helper (TinyMCE 6 / legacy / Atto / textarea) ---------- */
function getActiveEditor(id = 'id_contenteditor') {
    if (window.require) {
        try {
            const mgr = require('editor_tiny/editor');
            const inst = mgr.getInstanceForElementId(id);
            if (inst) return { type: 'tiny6', inst };
        } catch (e) {}
    }
    if (window.tinyMCE && tinyMCE.get(id)) {
        return { type: 'tinymce', inst: tinyMCE.get(id) };
    }
    if (window.Y?.M?.editor_atto?.Editor) {
        const inst = Y.M.editor_atto.Editor.getEditor(id);
        if (inst) return { type: 'atto', inst };
    }
    return { type: 'textarea', inst: document.getElementById(id) };
}

function setEditorContent(html, id = 'id_contenteditor') {
    const ed = getActiveEditor(id);
    console.log("HTML " + html)
    if (!ed.inst) return;
    switch (ed.type) {
        case 'tiny6':
        case 'tinymce':
            ed.inst.setContent(html);
            break;
        case 'atto':
            ed.inst.setHTML(html); ed.inst.fire('change');
            break;
        default:
            ed.inst.value = html;
    }
}

function getEditorContent(id = 'id_contenteditor') {
    const ed = getActiveEditor(id);
    if (!ed.inst) return '';
    switch (ed.type) {
        case 'tiny6':
        case 'tinymce':
            return ed.inst.getContent({ format: 'html' });
        case 'atto':
            return ed.inst.getHTML();
        default:
            return ed.inst.value;
    }
}

function whenEditorReady(cb, id = 'id_contenteditor', tries = 0) {
    const ed = getActiveEditor(id);
    if (ed.inst || tries > 40) { cb(); return; }
    setTimeout(() => whenEditorReady(cb, id, ++tries), 150);
}

/* ---------- Data & Logic ---------- */
const subtopicContents = {{#subtopicContentJson}}{{{.}}}{{/subtopicContentJson}}{{^subtopicContentJson}}{}{{/subtopicContentJson}};
const quizContents      = {{#quizContentJson}}{{{.}}}{{/quizContentJson}}{{^quizContentJson}}{}{{/quizContentJson}};
let currentSubtopicTitle = null;
let currentQuizTitle     = null;

/* ------------ Helpers for item id (from new logic) ------------ */
function setItemId(id) {
    const h = document.getElementById('contenteditor_itemid');
    if (h) h.value = id;
}
function getItemId() {
    const h = document.getElementById('contenteditor_itemid');
    return h ? Number(h.value) : 0;
}

/* ------------ Subtopic logic (new version) ------------ */
function loadSubtopicContent(title) {
    const obj = subtopicContents[title] || { text: '<p><em>No content available.</em></p>', itemid: 0 };
    whenEditorReady(() => {
        if(obj.text === undefined) {
            setEditorContent(obj);
        }
        else{
            setEditorContent(obj.text);
        }
        console.log('Set Editor content for:', title);
        setItemId(obj.itemid);
    });
    document.getElementById('content-title').innerText = title;
}

function saveCurrent() {
    if (!currentSubtopicTitle) return;
    subtopicContents[currentSubtopicTitle] = {
        text:   getEditorContent(),
        itemid: getItemId()
    };
}

/* ------------ Quiz logic (kept) ------------ */
function showQuizContent(title) {
    const quizData  = quizContents[title];
    const container = document.getElementById('quiz-content');
    container.innerHTML = '';
    currentQuizTitle = title;

    if (!quizData || !quizData.questions || !quizData.questions.length) {
        container.innerHTML = '<p><em>No questions found.</em></p>';
        return;
    }

    quizData.questions.forEach((q, index) => {
        const qDiv = document.createElement('div');
        qDiv.className = 'quiz-question mb-4 p-3 border rounded';
        qDiv.setAttribute('data-index', index);
        qDiv.innerHTML = `
            <p><strong>Q${index + 1}:</strong> <span class="question-text">${q.question}</span></p>
            <ul class="quiz-options">
                ${q.options.map(opt => `<li>${opt}</li>`).join('')}
            </ul>
            <p><em>Answer:</em> <span class="answer-text">${q.answer || q.correct_answer || ''}</span></p>
            <p><em>Explanation:</em> <span class="explanation-text">${q.explanation || ''}</span></p>
            <button type="button" class="btn btn-sm btn-outline-primary edit-question-btn">Edit</button>
            <button type="button" class="btn btn-sm btn-success save-question-btn d-none">Save</button>
        `;
        container.appendChild(qDiv);
    });

    document.getElementById('content-title').innerText = title;

    // Attach button events
    document.querySelectorAll('.edit-question-btn').forEach(btn => {
        btn.addEventListener('click', function () {
            const wrapper = this.closest('.quiz-question');
            const index   = wrapper.getAttribute('data-index');
            const data    = quizContents[title].questions[index];

            // Replace spans with inputs
            wrapper.querySelector('.question-text').outerHTML = `<input type="text" class="form-control mb-2 question-input" value="${data.question}">`;
            wrapper.querySelector('.quiz-options').outerHTML = `
                <div class="options-group">
                    ${data.options.map((opt, i) =>
                        `<input type="text" class="form-control mb-1 option-input" data-opt-index="${i}" value="${opt}">`).join('')}
                </div>
            `;
            wrapper.querySelector('.answer-text').outerHTML = `<input type="text" class="form-control mb-2 answer-input" value="${data.answer || data.correct_answer || ''}">`;
            wrapper.querySelector('.explanation-text').outerHTML = `<textarea class="form-control mb-2 explanation-input">${data.explanation || ''}</textarea>`;

            this.classList.add('d-none');
            wrapper.querySelector('.save-question-btn').classList.remove('d-none');
        });
    });

    document.querySelectorAll('.save-question-btn').forEach(btn => {
        btn.addEventListener('click', function () {
            const wrapper = this.closest('.quiz-question');
            const index   = wrapper.getAttribute('data-index');
            const inputs  = wrapper.querySelectorAll('.option-input');
           const original = quizContents[title].questions[index];
const updatedQuestion = {
    question_id:   original.question_id || 'q' + (parseInt(index) + 1),
    type:          original.type || 'multiple_choice',
    difficulty:    original.difficulty || 'easy',
    question:      wrapper.querySelector('.question-input').value.trim(),
    options:       Array.from(inputs).map(i => i.value.trim()),
    correct_answer: wrapper.querySelector('.answer-input').value.trim(),
    explanation:   wrapper.querySelector('.explanation-input').value.trim()
};


            // Save to quizContents
            quizContents[title].questions[index] = updatedQuestion;

            // Refresh the quiz preview with updated content
            showQuizContent(title);
        });
    });
}

function openQuizEditor(quizTitle) {
    const quizHtml  = quizContents[quizTitle] || '';
    const container = document.getElementById('quiz-question-editor-container');
    const tempDiv   = document.createElement('div');
    tempDiv.innerHTML = quizHtml;
    const questions = [...tempDiv.querySelectorAll('.question-block')];

    if (!questions.length) {
        container.innerHTML = '<p>No questions found.</p>';
        return;
    }

    container.innerHTML = '';
    questions.forEach((q, i) => {
        const block = document.createElement('div');
        block.className = 'mb-3 p-3 border rounded bg-light';
        block.innerHTML = `
            <label><strong>Question ${i + 1}</strong></label>
            <textarea class="form-control quiz-edit-textarea" data-index="${i}" rows="4">${q.innerHTML.trim()}</textarea>
        `;
        container.appendChild(block);
    });

    const modal = new bootstrap.Modal(document.getElementById('editQuizModal'));
    modal.show();

    document.getElementById('saveQuizEditsBtn').onclick = () => {
        const updated = Array.from(document.querySelectorAll('.quiz-edit-textarea'))
            .map(t => `<div class="question-block">${t.value.trim()}</div>`)
            .join('');

        quizContents[quizTitle] = updated;
        document.getElementById('quiz-content').innerHTML = updated;
        modal.hide();
    };
}

/* ------------ DOM Ready ------------ */
document.addEventListener('DOMContentLoaded', () => {
    // Load first subtopic by default (if present)
    const firstSub = document.querySelector('.subtopic-item[data-type="subtopic"]') || document.querySelector('.subtopic-item');
    if (firstSub) {
        currentSubtopicTitle = firstSub.dataset.subtitle;
        loadSubtopicContent(currentSubtopicTitle);
        // Ensure correct sections are visible
        const contentBody = document.getElementById('content-body');
        const quizDisplay = document.getElementById('quiz-display');
        if (contentBody) contentBody.style.display = '';
        if (quizDisplay)  quizDisplay.classList.add('d-none');
    }

    // Event delegation for item clicks (works even if items are re-rendered)
    document.addEventListener('click', (e) => {
        const item = e.target.closest('.subtopic-item');
        if (!item) return;

        const type  = item.dataset.type || 'subtopic';
        const title = item.dataset.subtitle;

        const contentBody = document.getElementById('content-body');
        const quizDisplay = document.getElementById('quiz-display');

        if (type === 'subtopic') {
            if (title === currentSubtopicTitle) return;
            // Leaving a quiz? no need to save quiz (edits handled separately), but save subtopic if needed
            saveCurrent();
            currentQuizTitle     = null;
            currentSubtopicTitle = title;

            // Toggle UI
            if (quizDisplay)  quizDisplay.classList.add('d-none');
            if (contentBody)  contentBody.style.display = '';

            loadSubtopicContent(title);

        } else if (type === 'quiz') {
            // Save current subtopic before switching
            saveCurrent();
            currentSubtopicTitle = null;

            if (contentBody) contentBody.style.display = 'none';
            if (quizDisplay)  quizDisplay.classList.remove('d-none');

            showQuizContent(title);
        }
    });

    // Quiz modal edit button
    document.getElementById('edit-quiz-btn')?.addEventListener('click', () => {
        if (currentQuizTitle) {
            openQuizEditor(currentQuizTitle);
        }
    });

    // Handle Save on Form Submit
    document.getElementById('step4-form').addEventListener('submit', () => {
        // Save subtopic currently in editor
        saveCurrent();

        // ✅ Save any active quiz edits into quizContents
        const quizBlocks = document.querySelectorAll('.quiz-question');
        quizBlocks.forEach(wrapper => {
            const index           = wrapper.getAttribute('data-index');
            const questionInput   = wrapper.querySelector('.question-input');
            const optionInputs    = wrapper.querySelectorAll('.option-input');
            const answerInput     = wrapper.querySelector('.answer-input');
            const explanationInput= wrapper.querySelector('.explanation-input');

            if (
                questionInput && optionInputs.length &&
                currentQuizTitle && quizContents[currentQuizTitle]?.questions
            ) {
               const original = quizContents[currentQuizTitle].questions[index];
               const updated = {
                              question_id:     original.question_id || 'q' + (parseInt(index) + 1),
                              type:            original.type || 'multiple_choice',
                              difficulty:      original.difficulty || 'easy',
                              question:        questionInput.value.trim(),
                              options:         Array.from(optionInputs).map(i => i.value.trim()),
                              correct_answer:  answerInput?.value.trim() || '',
                              explanation:     explanationInput?.value.trim() || ''
                             };

                quizContents[currentQuizTitle].questions[index] = updated;
            }
        });

        // Serialize to hidden inputs
        document.getElementById('topicsjson').value = JSON.stringify(subtopicContents);
        document.getElementById('quizjson').value   = JSON.stringify(quizContents);
        
    });
});
