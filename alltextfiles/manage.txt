            function buildTopicsFromString(string $topicsString, array $baseTopics): array
            {
                // Ensure UTF-8
                if (!mb_check_encoding($topicsString, 'UTF-8')) {
                    $topicsString = utf8_encode($topicsString);
                }

                // Strip control characters
                $topicsString = preg_replace('/[\x00-\x1F\x7F]/u', '', $topicsString);

                // Now decode safely
                $flat = json_decode($topicsString, true, 512, JSON_THROW_ON_ERROR);

                // $flat  = json_decode($topicsString, true, 512, JSON_THROW_ON_ERROR);
                $index = [];

                // 🔍 Build index from base topics
                foreach ($baseTopics as $topicIdx => $topic) {
                    foreach ($topic['subtopics'] as $subIdx => $sub) {
                        $key = mb_strtolower(trim($sub['title']));
                        $index[$key] = ['topicIdx' => $topicIdx, 'subIdx' => $subIdx];
                    }
                }
                $result = $baseTopics;
                $defaultTopicIdx = 0;

                // 🔁 Inject content into the result array
                foreach ($flat as $rawTitle => $rawContent) {
                    $cleanTitle = mb_strtolower(trim($rawTitle));

                    if (isset($index[$cleanTitle])) {
                        $t = $index[$cleanTitle]['topicIdx'];
                        $s = $index[$cleanTitle]['subIdx'];

                        $old = rtrim($result[$t]['subtopics'][$s]['content']);
                        $result[$t]['subtopics'][$s]['content'] = $old . "\n\n" . trim($rawContent);
                        $result[$t]['subtopics'][$s]['content_html'] = $GLOBALS['parsedown']->text($result[$t]['subtopics'][$s]['content']);
                    } else {
                        // New subtopic not matched, add it to default topic
                        if (preg_match('/<img|<iframe|<video|<div/i', $rawContent)) {
                            $content_html = trim($rawContent);
                        } else {
                            $content_html = $GLOBALS['parsedown']->text(trim($rawContent));
                        }

                        $newSub = [
                            'title'        => $rawTitle,
                            'content'      => trim($rawContent),
                            'examples'     => [],
                            'content_html' => $content_html,
                        ];

                        $result[$defaultTopicIdx]['subtopics'][] = $newSub;

                        $index[$cleanTitle] = [
                            'topicIdx' => $defaultTopicIdx,
                            'subIdx'   => count($result[$defaultTopicIdx]['subtopics']) - 1
                        ];

                    }
                }

                // ✅ Preserve quiz data from baseTopics
                // foreach ($result as $i => &$topic) {
                //     $title = $topic['title'] ?? "Untitled";
                //     if (isset($baseTopics[$i]['quiz_included'])) {
                //         $topic['quiz_included'] = $baseTopics[$i]['quiz_included'];
                //     }
                //     if (isset($baseTopics[$i]['quiz_data'])) {
                //         $topic['quiz_data'] = $baseTopics[$i]['quiz_data'];
                //     }
                // }

                foreach ($topics as &$topic) {
                    $title = $topic['title'] ?? '';

                    if (isset($quizcontents[$title])) {
                        $topic['quiz_included'] = true;
                        $topic['quiz_data'] = $quizcontents[$title];
                    }
                }
                 return $result;
            }
