<script>
document.addEventListener('DOMContentLoaded', function () {
    const topicsWrapper = document.getElementById('topics-wrapper');
    const topicOrderInput = document.getElementById('topic-order');
    const addTopicBtn = document.getElementById('add-topic-btn');
    const form = document.getElementById('aicourse-form');
    const saveBtn = document.getElementById('save-btn');
    const backBtn = form.querySelector('button[name="action"][value="back"]');

    const editModal = document.getElementById('editTopicModal');
    const editTitleInput = document.getElementById('edit-topic-title');
    const editDescriptionInput = document.getElementById('edit-topic-description');
    const editDurationSpan = document.getElementById('edit-topic-duration');
    const objectivesList = document.getElementById('edit-topic-objectives');
    const editIdInput = document.getElementById('edit-topic-id');

    const quizYesRadio = document.getElementById('quiz-yes');
const quizNoRadio = document.getElementById('quiz-no');
const quizQuestionCountGroup = document.getElementById('quiz-question-count-group');
const quizQuestionInput = document.getElementById('edit-topic-quiz-questions');


    const spinnerHTML = `
        <span class="spinner-border spinner-border-sm me-2" role="status" aria-hidden="true"></span>
        Generating..
    `;

    function toggleQuizQuestionCount() {
  if (quizYesRadio.checked) {
    quizQuestionCountGroup.style.display = 'block';
  } else {
    quizQuestionCountGroup.style.display = 'none';
  }
}

quizYesRadio.addEventListener('change', toggleQuizQuestionCount);
quizNoRadio.addEventListener('change', toggleQuizQuestionCount);


    // ===== Handle topic actions =====
    topicsWrapper.addEventListener('click', (e) => {
        const item = e.target.closest('.topic-item');
        if (!item) return;

        const deleteBtn = e.target.closest('.delete-btn');
        const editBtn = e.target.closest('.edit-btn');

        if (deleteBtn) {
            item.remove();
            updateTopicOrder();
        } else if (editBtn) {
            // Parse topicData from dataset
            let topicData = {};
            try {
                const rawData = item.dataset.topicdata || '{}';
                const decoded = decodeURIComponent(rawData);
                console.log('Raw data-topicdata:', rawData);
                console.log('Decoded data-topicdata:', decoded);
                topicData = JSON.parse(decoded);
                console.log('Parsed topicData:', topicData);
            } catch (err) {
                console.warn('Failed to parse topic data:', err);
            }

            // Populate modal inputs
            editIdInput.value = topicData.id || item.dataset.id || '';
            editTitleInput.value = topicData.title || '';
            editDescriptionInput.value = topicData.description || '';
            editDurationSpan.textContent = topicData.estimated_duration || '';

            if (topicData.has_quiz) {
  quizYesRadio.checked = true;
  quizQuestionCountGroup.style.display = 'block';
  quizQuestionInput.value = topicData.quiz_question_count || 1;
} else {
  quizNoRadio.checked = true;
  quizQuestionCountGroup.style.display = 'none';
}


            objectivesList.innerHTML = '';
            (topicData.learning_objectives || []).forEach(obj => {
                const li = document.createElement('li');
                li.textContent = obj;
                objectivesList.appendChild(li);
            });

            window.currentEditingItem = item;
            saveBtn.disabled = true;
            const modalInstance = new bootstrap.Modal(editModal);
            modalInstance.show();
        }
    });

    // ===== Modal Save =====
    document.getElementById('edit-topic-form').addEventListener('submit', function (e) {
        e.preventDefault();

        const newTitle = editTitleInput.value.trim();
        const newDesc = editDescriptionInput.value.trim();
        const duration = editDurationSpan.textContent.trim();
        const objectives = Array.from(objectivesList.querySelectorAll('li')).map(li => li.textContent.trim());

        const item = window.currentEditingItem;
        if (!item || !newTitle) return;

        const span = item.querySelector('span');
        const index = Array.from(topicsWrapper.querySelectorAll('.topic-item')).indexOf(item);
        span.textContent = `Topic ${index + 1}: ${newTitle}`;

        const topicData = {
            id: item.dataset.id,
            title: newTitle,
            description: newDesc,
            estimated_duration: duration,
            learning_objectives: objectives,
            has_quiz: quizYesRadio.checked,
            quiz_question_count: quizYesRadio.checked ? parseInt(quizQuestionInput.value || '1', 10) : 0,

            content: item.dataset.content || ''
        };

        item.dataset.topicdata = encodeURIComponent(JSON.stringify(topicData));

        updateTopicOrder();
        bootstrap.Modal.getInstance(editModal).hide();
        saveBtn.disabled = false;
    });

    // ===== Modal Close =====
    editModal.addEventListener('hidden.bs.modal', function () {
        saveBtn.disabled = false;
    });

    // ===== Add New Topic =====
    addTopicBtn.addEventListener('click', () => {
        const newIndex = topicsWrapper.querySelectorAll('.topic-item').length;

        const topicData = {
            id: `topic-${newIndex}`,
            title: 'New Topic',
            description: '',
            estimated_duration: '',
            learning_objectives: [],
            content: ''
        };

        const encodedData = encodeURIComponent(JSON.stringify(topicData));

        const newTopic = `
            <div class="topic-item"
                draggable="true"
                data-index="${newIndex}"
                data-id="topic-${newIndex}"
                data-content=""
                data-topicdata="${encodedData}"
                tabindex="0">
                <span>Topic ${newIndex + 1}: New Topic</span>
                <div>
                    <button type="button" class="move-btn"><i class="fas fa-arrows-alt"></i></button>
                    <button type="button" class="edit-btn"><i class="fas fa-pencil-alt"></i></button>
                    <button type="button" class="delete-btn"><i class="fas fa-trash"></i></button>
                </div>
            </div>
        `;

        topicsWrapper.insertAdjacentHTML('beforeend', newTopic);
        updateTopicOrder();
    });

    // ===== Drag and Drop Reordering =====
    topicsWrapper.addEventListener('dragstart', (e) => {
        const item = e.target.closest('.topic-item');
        if (item && item.dataset.index !== undefined) {
            e.dataTransfer.setData('text/plain', item.dataset.index);
        }
    });

    topicsWrapper.addEventListener('dragover', (e) => e.preventDefault());

    topicsWrapper.addEventListener('drop', (e) => {
        e.preventDefault();
        const fromIndex = e.dataTransfer.getData('text/plain');
        const toItem = e.target.closest('.topic-item');
        if (!toItem || fromIndex === toItem.dataset.index) return;

        const topics = Array.from(topicsWrapper.querySelectorAll('.topic-item'));
        const dragged = topics[fromIndex];
        topicsWrapper.insertBefore(dragged, toItem);
        updateTopicOrder();
    });

    // ===== Keyboard navigation for topic movement =====
    topicsWrapper.addEventListener('keydown', (e) => {
        const item = e.target.closest('.topic-item');
        if (!item) return;

        if (e.key === 'ArrowUp' && item.previousElementSibling) {
            topicsWrapper.insertBefore(item, item.previousElementSibling);
            updateTopicOrder();
        } else if (e.key === 'ArrowDown' && item.nextElementSibling) {
            topicsWrapper.insertBefore(item.nextElementSibling, item);
            updateTopicOrder();
        }
    });

    // ===== Update the hidden topic order field =====
    function updateTopicOrder() {
        const topics = Array.from(topicsWrapper.querySelectorAll('.topic-item'));
        const order = topics.map((item, index) => {
            const span = item.querySelector('span');
            const title = span.textContent.split(':').slice(1).join(':').trim();
            item.dataset.index = index;
            span.textContent = `Topic ${index + 1}: ${title}`;

            let topicData = {};
            try {
                topicData = JSON.parse(decodeURIComponent(item.dataset.topicdata || '{}'));
            } catch (e) {
                topicData = {
                    id: item.dataset.id || `topic-${index}`,
                    title: title,
                    description: '',
                    estimated_duration: '',
                    learning_objectives: [],
                    content: item.dataset.content || ''
                };
            }

            return {
                id: topicData.id,
                title: topicData.title,
                description: topicData.description,
                estimated_duration: topicData.estimated_duration,
                learning_objectives: topicData.learning_objectives,
                content: topicData.content
            };
        });

        try {
            topicOrderInput.value = JSON.stringify(order);
            console.log('Updated topic_order:', topicOrderInput.value);
        } catch (error) {
            console.error('Failed to stringify topic order:', error);
            topicOrderInput.value = '[]';
            alert('Error updating topics. Please try again.');
        }
    }

    updateTopicOrder();

    // ===== Handle form submission UI changes =====
    form.addEventListener('submit', function (e) {
        const submitter = e.submitter || document.activeElement;

        if (submitter && submitter === saveBtn) {
            setTimeout(() => {
                saveBtn.disabled = true;
                backBtn.disabled = true;
                addTopicBtn.disabled = true;
                saveBtn.innerHTML = spinnerHTML;
            }, 0);
        }
    });
});
</script>
